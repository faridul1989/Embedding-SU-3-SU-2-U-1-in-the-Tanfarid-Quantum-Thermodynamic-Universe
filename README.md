\begin{abstract}
We embed the Standard Model into the Tanfarid Quantum--Thermodynamic Universe (TQTU) by deriving SU(3)$\times$SU(2)$\times$U(1) gauge fields as collective modes of a plasma--spin substrate. Fermions are topological defects with masses $m_i = m_0 \varphi^{-n_i}$ and mixings from γ-defect angles. Predictions: $|V_{us}| = 0.225 \pm 0.002$, $\theta_{12} = \arcsin(1/\varphi)$, and 3 defect-mass peaks at 0.5 MeV, 106 MeV, 1.8 GeV in Be$^+$ φ-chains.
\end{abstract}
\keywords{Standard Model, topological defects, golden ratio, entropy, TQTU}

\section{Introduction}
TQTU claims that SU(3)$\times$SU(2)$\times$U(1) emerges from entropy-plasma defects, not fundamental fields. Here we construct the effective Lagrangian and derive masses and mixings.

\section{Gauge Fields as Collective Modes}
Define effective vector fields
\begin{equation}
A^a_\mu = \langle s_\mu \rangle_{\text{defect}}, \quad a = 1,\dots,8\text{ (SU(3))}; \quad W^i_\mu = \langle s_\mu \rangle_{\text{defect}}, \quad i = 1,2,3\text{ (SU(2))}; \quad B_\mu = \langle s_\mu \rangle_{\text{defect}}\text{ (U(1))}.
\end{equation}
Yang–Mills kinetic terms emerge as
\begin{equation}
\mathcal{L}_{\text{gauge}} = -\frac{1}{4} \sum_{a} F^{a}_{\mu\nu} F^{a\mu\nu}, \quad F^{a}_{\mu\nu} = \partial_\mu A^a_\nu - \partial_\nu A^a_\mu + g f^{abc} A^b_\mu A^c_\nu.
\end{equation}

\section{Fermions as Topological Defects}
Fermions are localized “spin–entropy knots” with topological charge $Q = \varphi^n$:
\begin{equation}
m_i = m_0 \varphi^{-n_i}, \quad n_i = 0,1,2 \text{ for } e,\mu,\tau.
\end{equation}
CKM matrix:
\begin{equation}
V_{ij} = \cos(\theta_{ij}), \quad \theta_{ij} = \gamma\text{-defect angle}.
\end{equation}
Predictions:
\begin{equation}
|V_{us}| = \cos(\arcsin(1/\varphi)) = 0.225 \pm 0.002.
\end{equation}
